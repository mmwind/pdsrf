124  BaseTreeNode* learn(BaseTreeNode* newNode){
125 
126  // Check for Root node
127  if(newNode == NULL){
128  newNode = addRoot();
129  }
130 
131  if(isVerbose())
132  INFO( getNodeTypeString(newNode->getNodeType()));
133 
134  // Initialize partition statistics
135  newNode->setNumClasses(this->getNumClasses());
136  newNode->setNumFeatures(this->getNumFeatures());
137 
138  // Check for Root node
139  if(newNode->isRoot()){
140  newNode->getStatistics()->copyStatistics( *getTrainDataset()->getStatistics() );
141  for(IndexType i=0; i<this->getNumTrainSamples(); i++)
142  newNode->addIndex(i);
143  }
144 
145  if(isVerbose())
146  INFO( "PointsDistr: " << newNode->getStatistics()->getCountsVector()->transpose() );
147 
148 
149  // Check if current subset of samples have the same class
150  // if so we need to produce terminal node with this class
151  if(newNode->getStatistics()->isUniform()){
152  newNode->computeLabel();
153  if(isVerbose()){
154  INFO("[UNIFORM] Terminal node #" << newNode->getId() << " with label " << newNode->getClassLabel() );
155  }
156  return(NULL);
157  }
158 
159  // Check stoppage criterion
160  if( getStoppageCriterion()->IsStoppageNeeded(newNode, *newNode->getStatistics()->getCountsVector()) ){
161  newNode->computeLabel();
162  if(isVerbose()){
163  INFO("[STOP] Terminal node #" << newNode->getId() << " with label " << newNode->getClassLabel() );
164  }
165  // produce terminal node
166  return(NULL);
167  }
168 
169  DataType bestSplitScore = MAX_DATA_VAL;
170  IndexType bestSplitIndex = 0;
171 
172  // Repeat until split with non-empty partitions is found
173  while (bestSplitScore == MAX_DATA_VAL){
174  getSplitGenerator()->clear();
175  getSplitGenerator()->generate(newNode->getStatistics()->getRanges());
176 
177  // For all generated splits
178  for(IndexType j = 0; j<getSplitGenerator()->getSplitNum(); j++ ){
179  BinarySplit* splitCandidate = getSplitGenerator()->getSplit(j);
180  PartitionStatistics lp( getNumClasses(), getNumFeatures() );
181  PartitionStatistics rp( getNumClasses(), getNumFeatures() );
182 
183  // For all points in this partition
184  for(IndexType i=0; i<newNode->getIndexes()->size(); i++ ){
185  IndexType idx = newNode->getIndexes()->at(i);
186  Sample *s = this->getTrainDataset()->getSample(idx);
187  if( splitCandidate->decide( s->getVector() ) ){
188  rp.addPoint(s);
189  } else {
190  lp.addPoint(s);
191  }
192  }
193 
194  DataType score = this->getSplitQualityMeasurer()->getScore( &lp, &rp );
195  splitCandidate->setScore(score);
196  if(bestSplitScore > score){
197  bestSplitScore = score;
198  bestSplitIndex = j;
199  }
200  }
201  }
202 
203 
204  /* Add new nodes and calculate statistics */
205  BaseTreeNode* nodeLeft = addNode(newNode, NODE_LEFT );
206  BaseTreeNode* nodeRight = addNode(newNode, NODE_RIGHT );
207  BinarySplit* best = getSplitGenerator()->getSplit(bestSplitIndex);
208 
209  PartitionStatistics lp( getNumClasses(), getNumFeatures() );
210  PartitionStatistics rp( getNumClasses(), getNumFeatures() );
211 
212  for(IndexType i=0; i<newNode->getIndexes()->size(); i++ ){
213  IndexType idx = newNode->getIndexes()->at(i);
214  Sample *s = this->getTrainDataset()->getSample(idx);
215  if( best->decide( s->getVector() ) ){
216  rp.addPoint(s);
217  nodeRight->addIndex(idx);
218  } else {
219  lp.addPoint(s);
220  nodeLeft->addIndex(idx);
221  }
222  }
223  nodeLeft->getStatistics()->copyStatistics(lp);
224  nodeRight->getStatistics()->copyStatistics(rp);
225  newNode->setFeatureIndex(best->getFeatureIndex());
226  newNode->setFeatureValue(best->getFeatureValue());
227  newNode->setScore(bestSplitScore);
228 
229  if(isVerbose()){
230  INFO("New split " << newNode->getId() << " at depth " << newNode->getDepth() << " with score " << newNode->getScore() );
231  INFO("Feature " << this->getTrainDataset()->getAttribute(newNode->getFeatureIndex())->getName() << " value " << newNode->getFeatureValue() );
232  }
233 
234  learn(nodeRight);
235  learn(nodeLeft);
236 
237  newNode->clearIndexes();
238  return(newNode);
239  }
