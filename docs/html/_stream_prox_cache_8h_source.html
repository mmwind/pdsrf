418  DataVector scale = *dataset->getStatistics()->getMaxRanges() - *dataset->getStatistics()->getMinRanges();
419  for(IndexType i=0; i< dataset->getNumSamples(); i++){
420  DataVector d = *dataset->getSample(i)->getVector() - *s->getVector();
